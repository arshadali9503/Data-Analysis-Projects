# listed_in column has multiple genres like:
# "Documentaries, International Movies"
# We need to SPLIT them and count each separately

from collections import Counter

# Split all genres and put in one big list
all_genres = []
for genres in df['listed_in']:
    genre_list = genres.split(',')
    for genre in genre_list:
        all_genres.append(genre.strip())

# Count each genre
genre_counts = Counter(all_genres)

# Convert to dataframe for easy plotting
genre_df = pd.DataFrame(genre_counts.most_common(10),
                        columns=['Genre', 'Count'])
print(genre_df)

# Create Bar Chart
plt.figure(figsize=(14, 7))

bars = plt.barh(genre_df['Genre'],
                genre_df['Count'],
                color=['#E50914', '#C0392B', '#922B21', '#7B241C',
                       '#641E16', '#B03A2E', '#A93226', '#E74C3C',
                       '#EC7063', '#F1948A'])

# Add value labels
for bar in bars:
    width = bar.get_width()
    plt.text(width, bar.get_y() + bar.get_height()/2.,
             f'{int(width):,}',
             ha='left', va='center',
             fontsize=11, fontweight='bold')

plt.title('Top 10 Genres on Netflix',
          fontsize=18,
          fontweight='bold')

plt.xlabel('Number of Titles', fontsize=13)
plt.ylabel('Genre', fontsize=13)
